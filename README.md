## –û–±—É—á–µ–Ω–∏–µ –º–æ–¥–µ–ª–µ–π VGG16 –∏ ResNet50
–ù–∞—Å—Ç—Ä–æ–π–∫–∞ –∏ –æ–±—É—á–µ–Ω–∏–µ –¥–≤—É—Ö –º–æ–¥–µ–ª–µ–π —Å –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º –ø—Ä–µ–¥–æ–±—É—á–µ–Ω–Ω—ã—Ö –≤–µ—Å–æ–≤ –∏–∑ ImageNet.

VGG16 ‚Äî —ç—Ç–æ –∞—Ä—Ö–∏—Ç–µ–∫—Ç—É—Ä–∞ —Å–≤–µ—Ä—Ç–æ—á–Ω–æ–π –Ω–µ–π—Ä–æ–Ω–Ω–æ–π —Å–µ—Ç–∏ (CNN), —Ä–∞–∑—Ä–∞–±–æ—Ç–∞–Ω–Ω–∞—è –≥—Ä—É–ø–ø–æ–π –∏—Å—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª–µ–π –∏–∑ –û–∫—Å—Ñ–æ—Ä–¥—Å–∫–æ–≥–æ —É–Ω–∏–≤–µ—Ä—Å–∏—Ç–µ—Ç–∞ –≤ —Ä–∞–º–∫–∞—Ö —Ä–∞–±–æ—Ç—ã –Ω–∞–¥ –ø—Ä–æ–µ–∫—Ç–æ–º Visual Geometry Group (VGG).

```
from tensorflow.keras.applications import VGG16, ResNet50
from tensorflow.keras import Model, layers, optimizers

# VGG16
base_model_vgg = VGG16(weights="imagenet", include_top=False, input_shape=(32, 32, 3)) # –ó–∞–≥—Ä—É–∂–∞–µ–º –±–∞–∑–æ–≤—É—é –º–æ–¥–µ–ª—å VGG16 –±–µ–∑ –ø–æ–ª–Ω–æ—Å–≤—è–∑–Ω–æ–≥–æ —Å–ª–æ—è.
x = layers.Flatten()(base_model_vgg.output) # –†–∞–∑–≤–æ—Ä–∞—á–∏–≤–∞–µ–º –≤—ã—Ö–æ–¥ –±–∞–∑–æ–≤–æ–π –º–æ–¥–µ–ª–∏ –≤ –ø–ª–æ—Å–∫–∏–π –º–∞—Å—Å–∏–≤.
x = layers.Dense(128, activation="relu")(x) # –î–æ–±–∞–≤–ª—è–µ–º –ø–ª–æ—Ç–Ω—ã–π —Å–ª–æ–π —Å 128 –Ω–µ–π—Ä–æ–Ω–∞–º–∏ –∏ –∞–∫—Ç–∏–≤–∞—Ü–∏–µ–π ReLU.
output = layers.Dense(43, activation="softmax")(x) # –î–æ–±–∞–≤–ª—è–µ–º –≤—ã—Ö–æ–¥–Ω–æ–π —Å–ª–æ–π —Å 43 –Ω–µ–π—Ä–æ–Ω–∞–º–∏ (–∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∫–ª–∞—Å—Å–æ–≤) –∏ –∞–∫—Ç–∏–≤–∞—Ü–∏–µ–π softmax.
model_vgg = Model(inputs=base_model_vgg.input, outputs=output) # –°–æ–∑–¥–∞–µ–º –º–æ–¥–µ–ª—å –Ω–∞ –æ—Å–Ω–æ–≤–µ –±–∞–∑–æ–≤–æ–π –º–æ–¥–µ–ª–∏ –∏ –¥–æ–±–∞–≤–ª–µ–Ω–Ω—ã—Ö —Å–ª–æ–µ–≤.
model_vgg.compile(optimizer=optimizers.Adam(learning_rate=0.001), loss="categorical_crossentropy", metrics=["accuracy"]) # –ö–æ–º–ø–∏–ª–∏—Ä—É–µ–º –º–æ–¥–µ–ª—å VGG16 (–û–ø—Ç–∏–º–∏–∑–∞—Ç–æ—Ä Adam —Å –Ω–∞—á–∞–ª—å–Ω—ã–º–∏ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º–∏),–§—É–Ω–∫—Ü–∏—è –ø–æ—Ç–µ—Ä—å –¥–ª—è –∑–∞–¥–∞—á–∏ –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏, –ú–µ—Ç—Ä–∏–∫–∞ –¥–ª—è –æ—Ü–µ–Ω–∫–∏ —Ç–æ—á–Ω–æ—Å—Ç–∏.
history_vgg = model_vgg.fit(small_X_train, small_y_train, validation_data=(X_val, y_val), epochs=5, batch_size=32) # –û–±—É—á–∞–µ–º –º–æ–¥–µ–ª—å VGG16 –Ω–∞ —É–º–µ–Ω—å—à–µ–Ω–Ω–æ–π –≤—ã–±–æ—Ä–∫–µ, (–û–±—É—á–∞—é—â–∞—è –≤—ã–±–æ—Ä–∫–∞,,–í–∞–ª–∏–¥–∞—Ü–∏–æ–Ω–Ω–∞—è –≤—ã–±–æ—Ä–∫–∞,–í–∞–ª–∏–¥–∞—Ü–∏–æ–Ω–Ω–∞—è –≤—ã–±–æ—Ä–∫–∞, –†–∞–∑–º–µ—Ä –º–∏–Ω–∏-–±–∞—Ç—á–∞)

# ResNet50
base_model_resnet = ResNet50(weights="imagenet", include_top=False, input_shape=(32, 32, 3)) # –ó–∞–≥—Ä—É–∂–∞–µ–º –±–∞–∑–æ–≤—É—é –º–æ–¥–µ–ª—å ResNet50 –±–µ–∑ –ø–æ–ª–Ω–æ—Å–≤—è–∑–Ω–æ–≥–æ —Å–ª–æ—è.
x = layers.Flatten()(base_model_resnet.output)  # –†–∞–∑–≤–æ—Ä–∞—á–∏–≤–∞–µ–º –≤—ã—Ö–æ–¥ –±–∞–∑–æ–≤–æ–π –º–æ–¥–µ–ª–∏ –≤ –ø–ª–æ—Å–∫–∏–π –º–∞—Å—Å–∏–≤.
x = layers.Dense(128, activation="relu")(x) # –î–æ–±–∞–≤–ª—è–µ–º –ø–ª–æ—Ç–Ω—ã–π —Å–ª–æ–π —Å 128 –Ω–µ–π—Ä–æ–Ω–∞–º–∏ –∏ –∞–∫—Ç–∏–≤–∞—Ü–∏–µ–π ReLU.
```

# –ó–∞–¥–∞–Ω–∏–µ 2: –ù–µ—Ü–µ–ª–µ–≤–∞—è –∞—Ç–∞–∫–∞ —É–∫–ª–æ–Ω–µ–Ω–∏—è
**–¶–µ–ª—å:** –ü—Ä–æ–≤–µ—Ä–∏—Ç—å —É—Å—Ç–æ–π—á–∏–≤–æ—Å—Ç—å –æ–±—É—á–µ–Ω–Ω—ã—Ö –º–æ–¥–µ–ª–µ–π –∫ FGSM –∏ PGD –∞—Ç–∞–∫–∞–º.

1. –ê—Ç–∞–∫–∞ FGSM:

* –ú–µ—Ç–æ–¥ —Å–æ–∑–¥–∞—ë—Ç –∞—Ç–∞–∫—É—é—â–∏–µ –ø—Ä–∏–º–µ—Ä—ã, –¥–æ–±–∞–≤–ª—è—è –∫ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è–º –∏—Å–∫–∞–∂–µ–Ω–∏—è, –ø—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω—ã–µ –∑–Ω–∞–∫—É –≥—Ä–∞–¥–∏–µ–Ω—Ç–∞ —Ñ—É–Ω–∫—Ü–∏–∏ –ø–æ—Ç–µ—Ä—å.
* –ò—Å–ø–æ–ª—å–∑—É—é—Ç—Å—è –ø–µ—Ä–≤—ã–µ 1000 –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π –∏–∑ —Ç–µ—Å—Ç–æ–≤–æ–≥–æ –Ω–∞–±–æ—Ä–∞.
* –ü—Ä–æ–≤–µ—Ä—è–µ—Ç—Å—è –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç—å —Ç–æ—á–Ω–æ—Å—Ç–∏ –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏ –æ—Ç –∑–Ω–∞—á–µ–Ω–∏–π –ø–∞—Ä–∞–º–µ—Ç—Ä–∞ –∏—Å–∫–∞–∂–µ–Ω–∏–π (ùúñ) –≤ –¥–∏–∞–ø–∞–∑–æ–Ω–µ [1/255, 2/255, 3/255, ..., 80/255].

2. –ê—Ç–∞–∫–∞ PGD:

* –ú–µ—Ç–æ–¥ —è–≤–ª—è–µ—Ç—Å—è –∏—Ç–µ—Ä–∞—Ç–∏–≤–Ω—ã–º —É–ª—É—á—à–µ–Ω–∏–µ–º FGSM.
* –¢–∞–∫–∂–µ –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –¥–∏–∞–ø–∞–∑–æ–Ω ùúñ –∫–∞–∫ –≤ FGSM.
* –°—Ä–∞–≤–Ω–∏–≤–∞–µ—Ç—Å—è —Ç–æ—á–Ω–æ—Å—Ç—å –¥–≤—É—Ö –º–æ–¥–µ–ª–µ–π –ø—Ä–∏ —Ä–∞–∑–Ω—ã—Ö —É—Ä–æ–≤–Ω—è—Ö –∏—Å–∫–∞–∂–µ–Ω–∏–π.

3. –û—Ç—á—ë—Ç:

* –ó–∞–ø–æ–ª–Ω—è–µ—Ç—Å—è –¢–∞–±–ª–∏—Ü–∞ 2 —Å —Ç–æ—á–Ω–æ—Å—Ç—å—é –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è ùúñ.
* –ü–æ—Å—Ç—Ä–æ–µ–Ω—ã –≥—Ä–∞—Ñ–∏–∫–∏ —Ç–æ—á–Ω–æ—Å—Ç–∏ –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏ –≤ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –æ—Ç ùúñ.
* –û—Ç–æ–±—Ä–∞–∂–µ–Ω—ã –∞—Ç–∞–∫—É—é—â–∏–µ –ø—Ä–∏–º–µ—Ä—ã –¥–ª—è FGSM —Å –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º–∏ ùúñ = [1/255, 5/255, 10/255, 50/255, 80/255].

## –†–µ–∞–ª–∏–∑–∞—Ü–∏—è FGSM-–∞—Ç–∞–∫–∏
–≠—Ç–æ—Ç –±–ª–æ–∫ —Ä–µ–∞–ª–∏–∑—É–µ—Ç FGSM-–∞—Ç–∞–∫—É –¥–ª—è —Å–æ–∑–¥–∞–Ω–∏—è –∞—Ç–∞–∫—É—é—â–∏—Ö –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π.

```
def fgsm_attack(model, image, label, epsilon=0.01):  # –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Ñ—É–Ω–∫—Ü–∏—é –¥–ª—è –∞—Ç–∞–∫–∏ FGSM (Fast Gradient Sign Method).
    image = tf.convert_to_tensor([image], dtype=tf.float32)  # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ –≤ —Ç–µ–Ω–∑–æ—Ä —Ç–∏–ø–∞ float32.
    label = tf.convert_to_tensor([label], dtype=tf.float32)  # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –º–µ—Ç–∫—É –≤ —Ç–µ–Ω–∑–æ—Ä —Ç–∏–ø–∞ float32.

    with tf.GradientTape() as tape:  # –°–æ–∑–¥–∞–µ–º –∫–æ–Ω—Ç–µ–∫—Å—Ç –¥–ª—è –≤—ã—á–∏—Å–ª–µ–Ω–∏—è –≥—Ä–∞–¥–∏–µ–Ω—Ç–æ–≤.
        tape.watch(image)  # –£–∫–∞–∑—ã–≤–∞–µ–º, —á—Ç–æ –º—ã –±—É–¥–µ–º –æ—Ç—Å–ª–µ–∂–∏–≤–∞—Ç—å –∏–∑–º–µ–Ω–µ–Ω–∏—è –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è.
        prediction = model(image)  # –ü–æ–ª—É—á–∞–µ–º –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–µ –º–æ–¥–µ–ª–∏ –¥–ª—è –¥–∞–Ω–Ω–æ–≥–æ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è.
        loss = tf.keras.losses.categorical_crossentropy(label, prediction)  # –í—ã—á–∏—Å–ª—è–µ–º —Ñ—É–Ω–∫—Ü–∏—é –ø–æ—Ç–µ—Ä—å (–∫—Ä–æ—Å—Å-—ç–Ω—Ç—Ä–æ–ø–∏—é).

    gradient = tape.gradient(loss, image)  # –í—ã—á–∏—Å–ª—è–µ–º –≥—Ä–∞–¥–∏–µ–Ω—Ç –ø–æ—Ç–µ—Ä—å –ø–æ –æ—Ç–Ω–æ—à–µ–Ω–∏—é –∫ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—é.
    perturbation = epsilon * tf.sign(gradient)  # –°–æ–∑–¥–∞–µ–º –≤–æ–∑–º—É—â–µ–Ω–∏–µ, –ø—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω–æ–µ –≥—Ä–∞–¥–∏–µ–Ω—Ç—É.
    adversarial_image = tf.clip_by_value(image + perturbation, 0, 1)  # –î–æ–±–∞–≤–ª—è–µ–º –≤–æ–∑–º—É—â–µ–Ω–∏–µ –∏ –æ–±—Ä–µ–∑–∞–µ–º –∑–Ω–∞—á–µ–Ω–∏—è –≤ –ø—Ä–µ–¥–µ–ª–∞—Ö [0, 1].
    return adversarial_image[0]  # –í–æ–∑–≤—Ä–∞—â–∞–µ–º –∞—Ç–∞–∫–æ–≤–∞–Ω–Ω–æ–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ.

adv_image = fgsm_attack(model_vgg, X_test[0], y_test[0])  # –ü—Ä–∏–º–µ–Ω—è–µ–º FGSM-–∞—Ç–∞–∫—É –∫ –ø–µ—Ä–≤–æ–º—É –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—é —Ç–µ—Å—Ç–æ–≤–æ–π –≤—ã–±–æ—Ä–∫–∏.
plt.imshow(adv_image.numpy())  # –û—Ç–æ–±—Ä–∞–∂–∞–µ–º –∞—Ç–∞–∫–æ–≤–∞–Ω–Ω–æ–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ.
```

## –ü—Ä–æ–≤–µ—Ä–∫–∞ —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ—Å—Ç–∏ FGSM
–≠—Ç–æ—Ç –∫–æ–¥ —Ç–µ—Å—Ç–∏—Ä—É–µ—Ç —É—Å—Ç–æ–π—á–∏–≤–æ—Å—Ç—å –º–æ–¥–µ–ª–∏ ResNet50 –∫ FGSM-–∞—Ç–∞–∫–∞–º.

